# making a smaller df

trees_df = 
  trees_2015 |> 
  drop_na() |> 
  group_by(postcode, borough, health) |> 
  summarize(
    n_trees = n())

# joining the trees data with sf object

trees_zip =
  merge(zip_nyc, trees_df, 
                        by.x = "MODZCTA",
                        by.y = "postcode") |> 
  janitor::clean_names()

trees_nta = 
  merge(zip_nyc, trees_df_nta, 
                        by.x = "MODZCTA",
                        by.y = "postcode") |>
